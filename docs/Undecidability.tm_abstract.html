Require Import Setoid.

Set Implicit Arguments.

An abstract proof of the non existence of Turing machines for deciding the Halting problem

Section tm_abstract.

  Variables (tm txt : Set)
            (eval : tm -> txt -> txt -> Prop)
            (tt : txt) (tt_eq_dec : forall x : txt, { x = tt } + { x <> tt })
            (Heval : forall m t b1 b2, eval m t b1 -> eval m t b2 -> b1 = b2).

  Definition halts m t := exists t', eval m t t'.

  Variables (enc : tm -> txt) (* Encoding of tm *)
            (pair : txt -> txt -> txt).

  (* (* This is a universal tm machine *)
  
             (univ : tm)                (* A decoding machine for enc *)
             (Huniv : forall m t r, eval m t r <-> eval univ (pair (enc m) t) r).
             
  *)

  (* There is a duplicating machine, we can composes machines,
     there is a always true, always loop machine
     there is an if then else machine *)

  Variables (tm_dup : tm) (Hdup : forall t r, eval tm_dup t r <-> r = pair t t)
             (tm_loop : tm) (Hloop : forall t r, ~ eval tm_loop t r)
             (tm_true : tm) (Htrue : forall t r, eval tm_true t r <-> r = tt)
             
             (tm_comp : tm -> tm -> tm)
             (Hcomp : forall m1 m2 t1 t3, eval (tm_comp m1 m2) t1 t3 <-> exists t2, eval m1 t1 t2
                                                                                 /\ eval m2 t2 t3)

             (tm_ite : tm -> tm -> tm -> tm)
             (Hite : forall t r m1 m2 m3, eval (tm_ite m1 m2 m3) t r <-> exists b, eval m1 t b
                                                                                 /\ (b = tt /\ eval m2 t r
                                                                                  \/ b <> tt /\ eval m3 t r)).

  (* Supposing the existence of a Halting decider leads to contradiction 
  
     A halting deciders always stops on any pair (enc m,t) where
     enc m is the encoding of a tm
     and it return tt iff m halts on t
  
  *)

  Hypothesis (tm_halt : tm)
             (Hhalt1 : forall m t, halts tm_halt (pair (enc m) t))
             (Hhalt2 : forall m t, halts m t <-> eval tm_halt (pair (enc m) t) tt).


  Let tm_diag := tm_ite (tm_comp tm_dup tm_halt) tm_loop tm_true.

  Let Hdiag1 : ~ halts tm_diag (enc tm_diag).
  Proof.
    intros H; generalize H.
    intros (r & Hr).
    apply Hite in Hr.
    destruct Hr as (b & H0 & [ (_ & H2) | (H1 & _) ]).
    + revert H2; apply Hloop.
    + apply Hcomp in H0.
      destruct H0 as (t & H0 & H2).
      apply Hdup in H0; subst t.
      apply Hhalt2 in H.
      apply H1.
      revert H2 H.
      apply Heval.
  Qed.

  Let Hdiag2 : ~ ~ halts tm_diag (enc tm_diag).
  Proof.
    intros H.
    rewrite Hhalt2 in H.
    destruct (Hhalt1 tm_diag (enc tm_diag)) as (b & Hb).
    destruct (tt_eq_dec b) as [ E | E ].
    + subst; apply H; auto.
    + apply H, Hhalt2.
      exists tt.
      apply Hite.
      exists b; split.
      - apply Hcomp.
        exists (pair (enc tm_diag) (enc tm_diag)); split; auto.
        apply Hdup; trivial.
      - right; split; auto.
        apply Htrue; trivial.
  Qed.

  Theorem no_halting : False.
  Proof. tauto. Qed.

End tm_abstract.

Check no_halting.
